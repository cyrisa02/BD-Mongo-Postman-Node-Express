const {
    slots,
    slotCreate,
    slotBook,
    slotUpdate,
    slotDelete,
} = require ("../controllers/slot");

function slotRoute(app) {
    //Create
    app.post("/slotCreate", slotCreate);

    //Book
    app.post("/slotBook",  slotBook);

    //Update
    app.post("/slotUpdate",  slotUpdate);

    //Delete
    app.post("/slotDelete",  slotDelete);


}

module.exports = slotRoute;
